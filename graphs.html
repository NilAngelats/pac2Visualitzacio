# Calculem la freqncia de les diferents espcies
sps_counts = data['species'].value_counts()

# Les ordenem de menys a ms, de tal manera que el grfic mostra les freqncies de ms a menys en sentit horari.
sorted_sps = sps_counts.sort_values(ascending=True).index

colors = ['#f3bc00', '#00af82', '#0075a4']

# Representem el grfic
plt.pie(sps_counts[sorted_sps], labels=sorted_sps, autopct='%1.1f%%', startangle=90, colors=colors)
plt.title("Distribuci de les espcies de pingins en lArxiplag de Palmer (Antrtida)")
plt.show()

In [12]:
# Calculem la freqüència de les diferents espècies
sps_counts = data['species'].value_counts()

# Les ordenem de menys a més, de tal manera que el gràfic mostra les freqüències de més a menys en sentit horari.
sorted_sps = sps_counts.sort_values(ascending=True).index

colors = ['#f3bc00', '#00af82', '#0075a4']

# Representem el gràfic
plt.pie(sps_counts[sorted_sps], labels=sorted_sps, autopct='%1.1f%%', startangle=90, colors=colors)
plt.title("Distribució de les espècies de pingüins en l’Arxipèlag de Palmer (Antàrtida)")
plt.show()
